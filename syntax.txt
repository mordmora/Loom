
<Identifier>   ::= [a-zA-Z_][a-zA-Z0-9_]*
<Number>       ::= [0-9]+ ("." [0-9]+)?
<String>       ::= '"' [^"]* '"'
<Boolean>      ::= "true" | "false"

<Program>      ::= <Declaration>* EOF

<Declaration>  ::= <AttributeList>? ( <FunctionDecl> | <StructDecl> | <VarDeclStmt> )

<AttributeList>::= ( "#[" <Attribute> "]" )+
<Attribute>    ::= <Identifier> ( "(" <AttributeParams> ")" )?
<AttributeParams>::= <Identifier> "=" <Expression> ( "," <Identifier> "=" <Expression> )*

<StructDecl>   ::= "struct" <Identifier> "{" <StructField>* "}"
<StructField>  ::= <Identifier> ":" <Type> ( ",")?

<FunctionDecl> ::= "func" <Identifier> "(" <ParamList>? ")" ( "->" <Type> )? <Block>
<ParamList>    ::= <Param> ( "," <Param> )*
<Param>        ::= <AttributeList>? <Identifier> ":" <Type>

<Type>         ::= <Identifier> ( "<" <Type> ">" )?     
                 | "*" <Type>                          
                 | "[" <Expression>? "]" <Type>            

<Block>        ::= "{" <Statement>* "}"

<Statement>    ::= <AttributeList>? <BaseStatement>

<BaseStatement>::= <VarDeclStmt>
                 | <IfStmt>
                 | <ForStmt>
                 | <WhileStmt>
                 | <ReturnStmt>
                 | <BreakStmt>
                 | <ContinueStmt>
                 | <ExprStmt>
                 | <Block>

<VarDeclStmt>  ::= <Identifier> ":" <Type> ( "=" <Expression> )?
<IfStmt>       ::= "if" <Expression> <Block> ( "else" ( <IfStmt> | <Block> ) )?

<ForStmt>      ::= "for" <VarDeclStmt> ";" <Expression> ";" <ExprStmt> <Block>

<WhileStmt>    ::= "while" <Expression> <Block>

<ReturnStmt>   ::= "return" <Expression>?
<BreakStmt>    ::= "break" 
<ContinueStmt> ::= "continue"

<ExprStmt>     ::= <Expression> ( <AssignOp> <Expression> )?
<AssignOp>     ::= "=" | "+=" | "-=" | "*=" | "/="

<Expression>   ::= <PrefixOp>* <Primary> <PostfixOp>* ( <InfixOp> <Expression> )*

<Primary>      ::= <Identifier> 
                 | <Number> 
                 | <String> 
                 | <Boolean>
                 | "(" <Expression> ")"
                 | <ArrayLiteral>

<ArrayLiteral> ::= "{" <ArgList>? "}"

<PrefixOp>     ::= "-" | "*" | "&" | "!"

<PostfixOp>    ::= "[" <Expression> "]"         
                 | "(" <ArgList>? ")"           
                 | "." <Identifier>             
                 | "++" | "--"                    

<InfixOp>      ::= "+" | "-" | "*" | "/" | "==" | "!=" | "<" | ">" | "<=" | ">="
<ArgList>      ::= <Expression> ( "," <Expression> )*
